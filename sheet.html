    void *tmp = malloc(size);
    memcpy(tmp, a, size);
    memcpy(a, b, size);
    memcpy(b, tmp, size);

}

6. Coding Challenges

Challenge 1 – Reverse a linked list iteratively

c
Copy
struct llnode *reverse(struct llnode *head) {
    struct llnode *prev = NULL, *cur = head, *next;
    while (cur) {
        next = cur->next;
        cur->next = prev;
        prev = cur;
        cur = next;
    }
    return prev;
}

Challenge 2 – Generic max element

c
Copy
void *generic_max(void *arr, int n, size_t size,
                  int (*cmp)(const void *, const void *)) {
    char *base = arr;
    char *best = base;
    for (int i = 1; i < n; ++i) {
        char *cur = base + i * size;
        if (cmp(cur, best) > 0) best = cur;
    }
    return best;
}

Challenge 3 – Deep copy a wrapped list

c
Copy
struct llist *list_cp(const struct llist *src) {
    struct llist *dst = list_create();
    struct llnode *p = src->front;
    while (p) {
        list_insert_back(dst, p->data);
        p = p->next;
    }
    return dst;
}

7. One-pager “Dirty Dozen” Checklist

  1. Always free what you malloc.
  2. Use sizeof(TYPE) never magic numbers.
  3. Returning &local is undefined behavior.
  4. scanf("%s", buf) must have width: "%79s" for char buf[80].
  5. char *s = "hello"; → read-only literal; mutate with char s[] = "hello";.
  6. void * cannot be dereferenced; cast first.
  7. sizeof(arr) only gives size in the same scope where defined.
  8. Pointer subtraction gives distance in elements, not bytes.
  9. Stack grows downward, heap upward; don’t mix them up in diagrams.
  10. const int *p → data is read-only; int *const p → pointer is read-only.
  11. For ADTs: interface (.h) = declarations; implementation (.c) = definitions.
  12. Amortized doubling strategy: realloc when len == maxlen, double maxlen.

B. Merge-sort (full)

c
Copy
void merge(int *dest, const int *a, int len_a,
           const int *b, int len_b) {
    int i=0, j=0, k=0;
    while (i<len_a && j<len_b)
        dest[k++] = (a[i] < b[j]) ? a[i++] : b[j++];
    while (i<len_a) dest[k++] = a[i++];
    while (j<len_b) dest[k++] = b[j++];
}

void merge_sort(int *arr, int len) {
    if (len<=1) return;
    int mid=len/2;
    int *left = malloc(mid*sizeof(int));
    int *right = malloc((len-mid)*sizeof(int));
    for (int i=0;i<mid;++i) left[i]=arr[i];
    for (int i=mid;i<len;++i) right[i-mid]=arr[i];
    merge_sort(left, mid);
    merge_sort(right, len-mid);
    merge(arr, left, mid, right, len-mid);
    free(left); free(right);
}

C. BST search (recursive)

c
Copy
bool bst_search(const struct bstnode *root, int key) {
    if (!root) return false;
    if (key == root->data) return true;
    return (key < root->data) ? bst_search(root->left, key)
                              : bst_search(root->right, key);
}
